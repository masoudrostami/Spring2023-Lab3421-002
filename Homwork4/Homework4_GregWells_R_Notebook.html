library("readxl")
df<-read_excel("winequality.xlsx")
# Create the function.
getmode <- function(v) {
   uniqv <- unique(v)
   uniqv[which.max(tabulate(match(v, uniqv)))]
}

max(df$quality)
## [1] 9
min(df$quality)
## [1] 3
sprintf("Quality rating minimum: %s",min(df$quality))
## [1] "Quality rating minimum: 3"
sprintf("Quality rating maximum: %s",max(df$quality))
## [1] "Quality rating maximum: 9"
sprintf("Quality rating mode: %s",getmode(df$quality))

Hypothesis Testing: Is there a significant difference in alcohol content between red wine and white wine?

library("readxl")
df<-read_excel("winequality.xlsx")

#First look at the distribution visually
df_al_type<-df %>% select(type,alcohol)
ggplot(df_al_type, aes(x = as.factor(type), y = alcohol, fill = type)) +
    geom_boxplot() + geom_jitter(shape = 15,color = "gray",size=0.5)+ 
    labs(y = "alcohol (% by vol)", x = "type") +
    ggtitle("Alcohol Content by Type (red/white)") +theme_classic() + theme(legend.position = "none")

t.test(alcohol~as.factor(type),  data=df_al_type)  
## 
##  Welch Two Sample t-test
## 
## data:  alcohol by as.factor(type)
## t = -2.859, df = 3100.5, p-value = 0.004278
## alternative hypothesis: true difference in means between group red and group white is not equal to 0
## 95 percent confidence interval:
##  -0.15388669 -0.02868117
## sample estimates:
##   mean in group red mean in group white 
##            10.42298            10.51427

## [1] Answer: Reject the null hypothesis. There is a significant difference ( p<0.05 ) in alcohol content between red wine and white wine.

Hypothesis Testing: Is there a significant difference in pH levels between high-quality and low-quality wines?

library("readxl")
df<-read_excel("winequality.xlsx")

#First look at the original quality distribution visually

ggplot(df, aes(x = quality)) + geom_bar() + scale_x_continuous(breaks = 2:10) + 
    ggtitle("Distribution of Wine Quality Ratings") +theme_classic() + theme(legend.position = "none")

#Now create a new column based on that info
#lets call high quality a rating of 6 or more and low quality less than 6

df_qual<-transform(df,hl_quality=ifelse(quality<6,"low","high"))

ggplot(df_qual, aes(x = as.factor(hl_quality), y = pH, fill = hl_quality)) +
    geom_boxplot() + geom_jitter(shape = 15,color = "gray",size=0.5)+ 
    labs(y = "pH level", x = "quality") +
    ggtitle("pH level by Quality (high/low)") +theme_classic() + theme(legend.position = "none")

t.test(pH~as.factor(hl_quality),  data=df_qual)  
## 
##  Welch Two Sample t-test
## 
## data:  pH by as.factor(hl_quality)
## t = 1.5167, df = 4955.6, p-value = 0.1294
## alternative hypothesis: true difference in means between group high and group low is not equal to 0
## 95 percent confidence interval:
##  -0.001839161  0.014409930
## sample estimates:
## mean in group high  mean in group low 
##           3.220807           3.214522
print("Answer: Failure to reject the null hypothesis. There is not a statistically significant difference ( p>0.05 ) in pH levels between high-quality and low-quality wines.",quote=FALSE)
## [1] Answer: Failure to reject the null hypothesis. There is not a statistically significant difference ( p>0.05 ) in pH levels between high-quality and low-quality wines.
